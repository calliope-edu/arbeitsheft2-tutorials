let Tempo = 1000
let streicheln = 0
basic.forever(function () {
    Tempo = 1000 - 200 * streicheln
    if (Tempo >= 200) {
        basic.showIcon(IconNames.Heart)
        basic.pause(Tempo)
        basic.clearScreen()
        basic.pause(Tempo)
    } else {
        basic.showIcon(IconNames.Heart)
    }
})
