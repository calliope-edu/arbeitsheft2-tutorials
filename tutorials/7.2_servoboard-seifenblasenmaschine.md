basic.forever(function () {
    if (grove.measureInCentimeters(DigitalPin.C16) < 10) {
        stab_unten()
        basic.pause(3000)
        stab_oben()
        basic.pause(1000)
        ventilator_an()
        basic.pause(2000)
        ventilator_aus()
    }
